对象的声明和使用

// 声明对象
var obj = {};
// 赋值取值
obj.key1 = 'value1';
obj['key2'] = 'value2';
var v1 = obj['key1'];
var v2 = obj.key2;
// 循环
for(var key in obj){
    console.log(key + ': ' + obj[key]);
}
// 删除键值对
delete obj['key1']; // for Firefox, Chrome etc.
obj['key1'] = undefined; // for IE;

// js中的对象更接近于Java中的Map
Map<String, Object> m = new HashMap<String, Object>();
// 取值赋值
m.put("key1", "value1");
m.put("key2", "value2");
String v1 = (String) m.get("key1");
String v2 = (String) m.get("key2");
// 循环
for(String key: m.keySet()){
    System.out.println(key + ": " + m.get(key));
}
// 删除键值对
m.remove('key1')

函数的声明和使用

// 声明函数有三种方法，但是真正使用的只有两种
function method1(value){
	console.log('hello ' + value);
}
var method2 = function(obj){
	if(typeof obj == 'function'){
		obj('Javascript');
	} else {
		console.log('hello ' + obj);
	}
}

// 调用函数
method1('Ablesky');
//output>> hello Ablesky
method2(method1);

类的声明和使用

				

					


// 类的声明(js只能声明public的类属性)
function User(username, password){
    this.username = username;
    this.password = password;
}
User.prototype = {
    sayHi: function(){
        console.log('Hi, ' + this.username);
    },
    toString: function(){
        return 'username: ' + this.username;
    }
}
/* 本质上js并不是面向对象的语言
 * 因此js中并没有完整的类的概念
 * 但是前辈们在充分发挥主观能动性后
 * 用原型链的方式模拟了类的构造甚至是继承
 * 这种手法在前端UI中有着重要的应用(例如Ext)
 */

// 类的使用
var user1 = new User('u1', '1234');
user1.sayHi()
// output>> Hi, u1
console.log(user1)
// output>> username: u1
console.log(user1 instanceof User);
// output>> true


					

					


// 类的声明
public class User{
	
    public String username;
    public String password;

    pubic User(String username, String password){
        this.username = username;
        this.password = password;
    }

    public void sayHi(){
        System.out.println("Hi, " + this.username);
    }

    pubic String toString(){
        return "username: " + this.username;
    }
}

// 类的使用
User user1 = new User("u1", "1234");
user1.sayHi();
// output>> Hi, u1
System.out.println(user1);
// output>> username: u1
System.out.println(user1 instanceof User);
// output>> true;
